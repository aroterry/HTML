import pandas as pd
from nltk import collocations
from nltk.tokenize import word_tokenize
from collections import Counter
from wordcloud import WordCloud
import matplotlib.pyplot as plt
import networkx as nx
import nltk
from google.colab import files

nltk.download('punkt')

# Upload the product reviews dataset
uploaded = files.upload()

# Assuming the uploaded file is named 'product_reviews.csv'
data = pd.read_csv('Product Review.csv')  # Replace with your uploaded file name

# Concatenate all reviews into a single string
all_reviews = ' '.join(data['name'])

# Tokenize the text into words
words = word_tokenize(all_reviews.lower())

# Remove common words that might not offer much insight
stop_words = {'the', 'and', 'to', 'of', 'a', 'is', 'in', 'it', 'that', 'this'}
filtered_words = [word for word in words if word not in stop_words]

# Extract collocations
finder = collocations.BigramCollocationFinder.from_words(filtered_words)
collocations = {bigram: freq for bigram, freq in finder.ngram_fd.items()}

# Get the top 3 collocations
top_collocations = Counter(collocations).most_common(3)
for collocation, score in top_collocations:
    print(f'{collocation}: {score}')

# Visualize the collocations as a Word Cloud
text = ' '.join(['_'.join(collocation) for collocation, _ in top_collocations])
wordcloud = WordCloud(width=800, height=400, background_color='white').generate(text)

plt.figure(figsize=(10, 5))
plt.imshow(wordcloud, interpolation='bilinear')
plt.axis('off')
plt.title('Top Collocations in Product Reviews (Word Cloud)')
plt.show()

# Create a Network Graph of collocations
G = nx.Graph()
for collocation, score in collocations.items():
    word1, word2 = collocation
    G.add_edge(word1, word2, weight=score)

plt.figure(figsize=(10, 8))
pos = nx.spring_layout(G, k=0.3)
nx.draw(G, pos, with_labels=True, font_size=10, node_color='skyblue', node_size=1200, edge_color='gray',
        linewidths=1, alpha=0.7)
plt.title('Collocations Network Graph')
plt.show()
